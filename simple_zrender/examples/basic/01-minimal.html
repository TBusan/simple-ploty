最小示例

这是最简单的等值线图示例，使用 Marching Squares 算法计算等值线并用 ZRender 渲染。 

import * as zrender from 'zrender';
import { MarchingSquares, PathFinder } from 'zrender-contour';

// 生成数据
const data = {
  z: [[1,2,3],[2,3,4],[3,4,5]],
  x: [0, 1, 2],
  y: [0, 1, 2]
};

// 计算等值线
const ms = new MarchingSquares(data, [2, 3, 4]);
const levelInfo = ms.computeLevels();

// 追踪路径
for (const [level, info] of levelInfo) {
  const finder = new PathFinder(data, info);
  const result = finder.findAllPaths();
  // 渲染 result.paths...
}
